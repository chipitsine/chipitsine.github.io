257 * value is copied in <*buf>. Here is the encoding format:
258 *
259 * 0 <= X < 240 : 1 byte (7.875 bits) [ XXXX XXXX ]
260 * 240 <= X < 2288 : 2 bytes (11 bits) [ 1111 XXXX ] [ 0XXX XXXX ]
261 * 2288 <= X < 264432 : 3 bytes (18 bits) [ 1111 XXXX ] [ 1XXX XXXX ] [ 0XXX XXXX ]
262 * 264432 <= X < 33818864 : 4 bytes (25 bits) [ 1111 XXXX ] [ 1XXX XXXX ]*2 [ 0XXX XXXX ]
263 * 33818864 <= X < 4328786160 : 5 bytes (32 bits) [ 1111 XXXX ] [ 1XXX XXXX ]*3 [ 0XXX XXXX ]
264 * ...
265 *
266 * On success, it returns the number of written bytes and <*buf> is moved after
267 * the encoded value. Otherwise, it returns -1. */
268static inline int
269encode_varint(uint64_t i, char **buf, char *end)
270{
271 unsigned char *p = (unsigned char *)*buf;
272 int r;
273
274 if (p >= (unsigned char *)end)
275 return -1;
276
277 if (i < 240) {
278 *p++ = i;
279 *buf = (char *)p;
280 return 1;
281 }
282
283 *p++ = (unsigned char)i | 240;
284 i = (i - 240) >> 4;
285 while (i >= 128) {
286 if (p >= (unsigned char *)end)
287 return -1;
288 *p++ = (unsigned char)i | 128;
289 i = (i - 128) >> 7;
290 }
291
292 if (p >= (unsigned char *)end)
293 return -1;
294 *p++ = (unsigned char)i;
295
296 r = ((char *)p - *buf);
297 *buf = (char *)p;
298 return r;
299}

873/* Simple ffs implementation. It returns the position of the lowest bit set to
874 * one, starting at 1. It is illegal to call it with a==0 (undefined result).
875 */
876static inline unsigned int my_ffsl(unsigned long a)
877{
878 unsigned long cnt;
879
880#if defined(__x86_64__1)
881 __asm__("bsf %1,%0\n" : "=r" (cnt) : "rm" (a));
882 cnt++;
883#else
884
885 cnt = 1;
886#if LONG_MAX9223372036854775807L > 0x7FFFFFFFL /* 64bits */
887 if (!(a & 0xFFFFFFFFUL)) {
888 a >>= 32;
889 cnt += 32;
890 }
891#endif
892 if (!(a & 0XFFFFU)) {
893 a >>= 16;
894 cnt += 16;
895 }
896 if (!(a & 0XFF)) {
897 a >>= 8;
898 cnt += 8;
899 }
900 if (!(a & 0xf)) {
901 a >>= 4;
902 cnt += 4;
903 }
904 if (!(a & 0x3)) {
905 a >>= 2;
906 cnt += 2;
907 }
908 if (!(a & 0x1)) {
909 cnt += 1;
910 }
911#endif /* x86_64 */
912
913 return cnt;
914}
