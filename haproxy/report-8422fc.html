521 payload += strlen(PAYLOAD_PATTERN"<<");
522 }
523
524 /*
525 * Get pointers on words.
526 * One extra slot is reserved to store a pointer on a null byte.
527 */
528 while (i < MAX_STATS_ARGS64 && p < end) {
529 int j, k;
530
531 /* skip leading spaces/tabs */
532 p += strspn(p, " \t");
533 if (!*p)
534 break;
535
536 args[i] = p;
537 p += strcspn(p, " \t");

539
540 /* unescape backslashes (\) */
541 for (j = 0, k = 0; args[i][k]; k++) {
542 if (args[i][k] == '\\') {
543 if (args[i][k + 1] == '\\')
544 k++;
545 else
546 continue;
547 }
548 args[i][j] = args[i][k];
549 j++;
550 }
551 args[i][j] = 0;
552
553 i++;
554 }
555 /* fill unused slots */
556 p = appctx->chunk->area + appctx->chunk->data;
557 for (; i < MAX_STATS_ARGS64 + 1; i++)
558 args[i] = p;
559
560 kw = cli_find_kw(args);
561 if (!kw)
562 return 0;
563
564 /* in a worker or normal process, don't display master only commands */
565 if (master == 0 && (kw->level & ACCESS_MASTER_ONLY0x10))
566 return 0;
567
568 /* in master don't displays if we don't have the master bits */
569 if (master == 1 && !(kw->level & (ACCESS_MASTER_ONLY0x10|ACCESS_MASTER0x8)))
570 return 0;
571
572 /* only accept expert commands in expert mode */
573 if ((kw->level & ~appctx->cli_level) & ACCESS_EXPERT0x20)
574 return 0;
575
576 appctx->io_handler = kw->io_handler;
577 appctx->io_release = kw->io_release;
578
579 if (kw->parse && kw->parse(args, payload, appctx, kw->private) != 0)
580 goto fail;
581
582 /* kw->parse could set its own io_handler or io_release handler */
583 if (!appctx->io_handler)
584 goto fail;
585
586 appctx->st0 = CLI_ST_CALLBACK;
587 return 1;
588fail:
589 appctx->io_handler = NULL((void*)0);
590 appctx->io_release = NULL((void*)0);
591 return 1;
592}

626/* This I/O handler runs as an applet embedded in a stream interface. It is
627 * used to processes I/O from/to the stats unix socket. The system relies on a
628 * state machine handling requests and various responses. We read a request,
629 * then we process it and send the response, and we possibly display a prompt.
630 * Then we can read again. The state is stored in appctx->st0 and is one of the
631 * CLI_ST_* constants. appctx->st1 is used to indicate whether prompt is enabled
632 * or not.
633 */
634static void cli_io_handler(struct appctx *appctx)
635{
636 struct stream_interface *si = appctx->owner;
637 struct channel *req = si_oc(si);
638 struct channel *res = si_ic(si);
639 struct bind_conf *bind_conf = strm_li(si_strm(si))->bind_conf;
640 int reql;
641 int len;

645
646 /* Check if the input buffer is available. */
647 if (res->buf.size == 0) {

649 * already subscribed to a wait list to get a buffer.
650 */

659 appctx->cli_severity_output = bind_conf->severity_output;
660 appctx->st0 = CLI_ST_GETREQ;
661 appctx->cli_level = bind_conf->level;
662 }
663 else if (appctx->st0 == CLI_ST_END) {

665 * side, the conditions below will complete if needed.
666 */

670 }
671 else if (appctx->st0 == CLI_ST_GETREQ) {

673
674 /* use a trash chunk to store received data */

678 appctx->st0 = CLI_ST_END;
679 continue;
680 }
681 }
682
683 str = appctx->chunk->area + appctx->chunk->data;
684
685 /* ensure we have some output room left in the event we
686 * would want to return some info right after parsing.
687 */
688 if (buffer_almost_full(si_ib(si))) {
689 si_rx_room_blk(si);
690 break;
691 }

693 /* '- 1' is to ensure a null byte can always be inserted at the end */

701 }

705 * one, we replace it with an LF and skip only this part.

708 if (str[len] == '\\') {

711 }
712 if (str[len] == ';') {
713 str[len] = '\n';
714 reql = len + 1;
715 break;
716 }
717 }
718 }
719
720 /* now it is time to check that we have a full line,
721 * remove the trailing \n and possibly \r, then cut the
722 * line.

726 appctx->st0 = CLI_ST_END;
727 continue;
728 }
729
730 if (len && str[len-1] == '\r')

732
733 str[len] = '\0';
734 appctx->chunk->data += len;
735
736 if (appctx->st1 & APPCTX_CLI_ST1_PAYLOAD(1 << 1)) {
737 appctx->chunk->area[appctx->chunk->data] = '\n';
738 appctx->chunk->area[appctx->chunk->data + 1] = 0;
739 appctx->chunk->data++;
740 }

742 appctx->st0 = CLI_ST_PROMPT;
743
744 if (appctx->st1 & APPCTX_CLI_ST1_PAYLOAD(1 << 1)) {
745 /* empty line */
746 if (!len) {
747 /* remove the last two \n */
748 appctx->chunk->data -= 2;
749 appctx->chunk->area[appctx->chunk->data] = 0;
750
751 if (!cli_parse_request(appctx))
752 cli_gen_usage_msg(appctx);
753
754 chunk_reset(appctx->chunk);
755 /* NB: cli_sock_parse_request() may have put
756 * another CLI_ST_O_* into appctx->st0.
757 */
758
759 appctx->st1 &= ~APPCTX_CLI_ST1_PAYLOAD(1 << 1);
760 }
761 }
762 else {
763 /*
764 * Look for the "payload start" pattern at the end of a line
765 * Its location is not remembered here, this is just to switch
766 * to a gathering mode.
767 */
768 if (!strcmp(appctx->chunk->area + appctx->chunk->data - strlen(PAYLOAD_PATTERN"<<"), PAYLOAD_PATTERN"<<"))
769 appctx->st1 |= APPCTX_CLI_ST1_PAYLOAD(1 << 1);

811 }

815 appctx->st0 == CLI_ST_PRINT_DYN) {

817 appctx->ctx.cli.err = NULL((void*)0);
818 }
819 appctx->st0 = CLI_ST_PROMPT;
820 }
821 else
822 si_rx_room_blk(si);
823 break;
824
825 case CLI_ST_CALLBACK: /* use custom pointer */
826 if (appctx->io_handler)
827 if (appctx->io_handler(appctx)) {
828 appctx->st0 = CLI_ST_PROMPT;
829 if (appctx->io_release) {

841 if (appctx->st0 == CLI_ST_PROMPT) {
842 const char *prompt = "";
843
844 if (appctx->st1 & APPCTX_CLI_ST1_PROMPT(1 << 0)) {
845 /*
846 * when entering a payload with interactive mode, change the prompt
847 * to emphasize that more data can still be sent
848 */
849 if (appctx->chunk->data && appctx->st1 & APPCTX_CLI_ST1_PAYLOAD(1 << 1))
850 prompt = "+ ";
851 else
852 prompt = "\n> ";
853 }
854 else {

856 prompt = "\n";
857 }
858
859 if (ci_putstr(si_ic(si), prompt) != -1)
860 appctx->st0 = CLI_ST_GETREQ;
861 else
862 si_rx_room_blk(si);
863 }
864
865 /* If the output functions are still there, it means they require more room. */
866 if (appctx->st0 >= CLI_ST_OUTPUT)
867 break;
868
869 /* Now we close the output if one of the writers did so,
870 * or if we're not in interactive mode and the request
871 * buffer is empty. This still allows pipelined requests
872 * to be sent in non-interactive mode.
873 */
874 if (((res->flags & (CF_SHUTW0x00002000|CF_SHUTW_NOW0x00004000))) ||
875 (!(appctx->st1 & APPCTX_CLI_ST1_PROMPT(1 << 0)) && !co_data(req) && (!(appctx->st1 & APPCTX_CLI_ST1_PAYLOAD(1 << 1))))) {
876 appctx->st0 = CLI_ST_END;

878 }
879
880 /* switch state back to GETREQ to read next requests */
881 appctx->st0 = CLI_ST_GETREQ;
882 /* reactivate the \n at the end of the response for the next command */
883 appctx->st1 &= ~APPCTX_CLI_ST1_NOLF(1 << 2);

885 }

889 __FUNCTION__, __LINE__, req->flags, res->flags, si->state);
890 /* Other side has closed, let's abort if we have no more processing to do
891 * and nothing more to consume. This is comparable to a broken pipe, so
892 * we forward the close to the request side so that it flows upstream to
893 * the client.
894 */
895 si_shutw(si);
896 }
897
898 if ((req->flags & CF_SHUTW0x00002000) && (si->state == SI_ST_EST) && (appctx->st0 < CLI_ST_OUTPUT)) {
899 DPRINTF(stderr, "%s@%d: buf to si closed. req=%08x, res=%08x, st=%d\n",
900 __FUNCTION__, __LINE__, req->flags, res->flags, si->state);
901 /* We have no more processing to do, and nothing more to send, and
902 * the client side has closed. So we'll forward this state downstream
903 * on the response buffer.
904 */
905 si_shutr(si);
906 res->flags |= CF_READ_NULL0x00000001;
907 }
908
909 out:
910 DPRINTF(stderr, "%s@%d: st=%d, rqf=%x, rpf=%x, rqh=%lu, rqs=%lu, rh=%lu, rs=%lu\n",
911 __FUNCTION__, __LINE__,
912 si->state, req->flags, res->flags, ci_data(req), co_data(req), ci_data(res), co_data(res));
913}

2001/*
2002 * Parse the CLI request:
2003 * - It does basically the same as the cli_io_handler, but as a proxy
2004 * - It can exec a command and strip non forwardable commands
2005 *
2006 * Return:
2007 * - the number of characters to forward or
2008 * - 1 if there is an error or not enough data
2009 */
2010int pcli_parse_request(struct stream *s, struct channel *req, char **errmsg, int *next_pid)
2011{
2012 char *str = (char *)ci_head(req);
2013 char *end = (char *)ci_stop(req);
2014 char *args[MAX_STATS_ARGS64 + 1]; /* +1 for storing a NULL */
2015 int argl; /* number of args */
2016 char *p;
2017 char *trim = NULL((void*)0);
2018 char *payload = NULL((void*)0);
2019 int wtrim = 0; /* number of words to trim */
2020 int reql = 0;
2021 int ret;
2022 int i = 0;
2023
2024 p = str;
2025
2026 if (!(s->pcli_flags & PCLI_F_PAYLOAD0x8)) {
2027
2028 /* Looks for the end of one command */
2029 while (p+reql < end) {
2030 /* handle escaping */
2031 if (p[reql] == '\\') {
2032 reql++;
2033 continue;
2034 }
2035 if (p[reql] == ';' || p[reql] == '\n') {
2036 /* found the end of the command */
2037 p[reql] = '\n';
2038 reql++;
2039 break;
2040 }
2041 reql++;
2042 }
2043 } else {
2044 while (p+reql < end) {
2045 if (p[reql] == '\n') {
2046 /* found the end of the line */
2047 reql++;
2048 break;
2049 }
2050 reql++;
2051 }
2052 }
2053
2054 /* set end to first byte after the end of the command */
2055 end = p + reql;
2056
2057 /* there is no end to this command, need more to parse ! */
2058 if (*(end-1) != '\n') {
2059 return -1;
2060 }
2061
2062 if (s->pcli_flags & PCLI_F_PAYLOAD0x8) {
2063 if (reql == 1) /* last line of the payload */
2064 s->pcli_flags &= ~PCLI_F_PAYLOAD0x8;
2065 return reql;
2066 }
2067
2068 *(end-1) = '\0';
2069
2070 /* splits the command in words */
2071 while (i < MAX_STATS_ARGS64 && p < end) {
2072 int j, k;
2073
2074 /* skip leading spaces/tabs */
2075 p += strspn(p, " \t");
2076 if (!*p)
2077 break;
2078
2079 args[i] = p;
2080 p += strcspn(p, " \t");
2081 *p++ = 0;
2082
2083 /* unescape backslashes (\) */
2084 for (j = 0, k = 0; args[i][k]; k++) {
2085 if (args[i][k] == '\\') {
2086 if (args[i][k + 1] == '\\')
2087 k++;
2088 else
2089 continue;
2090 }
2091 args[i][j] = args[i][k];
2092 j++;
2093 }
2094 args[i][j] = 0;
2095 i++;
2096 }
2097
2098 argl = i;
2099
2100 for (; i < MAX_STATS_ARGS64 + 1; i++)
2101 args[i] = NULL((void*)0);
2102
2103 wtrim = pcli_find_and_exec_kw(s, args, argl, errmsg, next_pid);
2104
2105 /* End of words are ending by \0, we need to replace the \0s by spaces
21061 before forwarding them */
2107 p = str;
2108 while (p < end-1) {
2109 if (*p == '\0')
2110 *p = ' ';
2111 p++;
2112 }
2113
2114 payload = strstr(str, PAYLOAD_PATTERN"<<");
2115 if ((end - 1) == (payload + strlen(PAYLOAD_PATTERN"<<"))) {
2116 /* if the payload pattern is at the end */
2117 s->pcli_flags |= PCLI_F_PAYLOAD0x8;
2118 ret = reql;
2119 }
2120
2121 *(end-1) = '\n';
2122
2123 if (wtrim > 0) {
2124 trim = &args[wtrim][0];
2125 if (trim == NULL((void*)0)) /* if this was the last word in the table */
2126 trim = end;
2127
2128 b_del(&req->buf, trim - str);
2129
2130 ret = end - trim;
2131 } else if (wtrim < 0) {
2132 /* parsing error */
2133 return -1;
2134 } else {
2135 /* the whole string */
2136 ret = end - str;
2137 }
2138
2139 if (ret > 1) {
2140 if (pcli_has_level(s, ACCESS_LVL_ADMIN3)) {
2141 goto end;
2142 } else if (pcli_has_level(s, ACCESS_LVL_OPER2)) {
2143 ci_insert_line2(req, 0, "operator -", strlen("operator -"));
2144 ret += strlen("operator -") + 2;
2145 } else if (pcli_has_level(s, ACCESS_LVL_USER1)) {
2146 ci_insert_line2(req, 0, "user -", strlen("user -"));
2147 ret += strlen("user -") + 2;
2148 }
2149 }
2150end:
2151
2152 return ret;
2153}

30/* This function tries to temporarily disable a listener, depending on the OS
31 * capabilities. Linux unbinds the listen socket after a SHUT_RD, and ignores
32 * SHUT_WR. Solaris refuses either shutdown(). OpenBSD ignores SHUT_RD but
33 * closes upon SHUT_WR and refuses to rebind. So a common validation path
34 * involves SHUT_WR && listen && SHUT_RD. In case of success, the FD's polling
35 * is disabled. It normally returns non-zero, unless an error is reported.
36 */
37int pause_listener(struct listener *l);

39/* This function tries to resume a temporarily disabled listener.
40 * The resulting state will either be LI_READY or LI_FULL. 0 is returned
41 * in case of failure to resume (eg: dead socket).

45/* This function adds all of the protocol's listener's file descriptors to the
