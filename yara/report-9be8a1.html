249// file_queue is size-limited queue stored as a circular array, files are
250// removed from queue_head position and new files are added at queue_tail
251// position. The array has room for one extra element to avoid queue_head
252// being equal to queue_tail in a full queue. The only situation where
253// queue_head == queue_tail is when queue is empty.

255QUEUED_FILE file_queue[MAX_QUEUED_FILES64 + 1];

258int queue_tail;

269static int file_queue_init()
270{
271 int result;
272
273 queue_tail = 0;
274 queue_head = 0;
275
276 result = mutex_init(&queue_mutex);
277
278 if (result != 0)
279 return result;
280
281 result = semaphore_init(&used_slots, 0);
282
283 if (result != 0)
284 return result;
285
286 return semaphore_init(&unused_slots, MAX_QUEUED_FILES64);
287}

292 mutex_destroy(&queue_mutex);
293 semaphore_destroy(&unused_slots);
294 semaphore_destroy(&used_slots);
295}

307static void file_queue_put(
308 const char* file_path)
309{
310 semaphore_wait(&unused_slots);
311 mutex_lock(&queue_mutex);
312
313 file_queue[queue_tail].path = strdup(file_path);
314 queue_tail = (queue_tail + 1) % (MAX_QUEUED_FILES64 + 1);
315
316 mutex_unlock(&queue_mutex);
317 semaphore_release(&used_slots);
318}

320
321static char* file_queue_get()
322{
323 char* result;
324
325 semaphore_wait(&used_slots);
326 mutex_lock(&queue_mutex);
327
328 if (queue_head == queue_tail) // queue is empty
329 {
330 result = NULL((void*)0);
331 }
332 else
333 {
334 result = file_queue[queue_head].path;
335 queue_head = (queue_head + 1) % (MAX_QUEUED_FILES64 + 1);
336 }
337
338 mutex_unlock(&queue_mutex);
339 semaphore_release(&unused_slots);
340
341 return result;
342}

359static void scan_dir(
360 const char* dir,
361 int recursive,
362 time_t start_time)
363{
364 static char path_and_mask[MAX_PATH1024];
365
366 snprintf(path_and_mask, sizeof(path_and_mask), "%s\\*", dir);
367
368 WIN32_FIND_DATA FindFileData;
369 HANDLE hFind = FindFirstFile(path_and_mask, &FindFileData);
370
371 if (hFind != INVALID_HANDLE_VALUE)
372 {
373 do
374 {
375 char full_path[MAX_PATH1024];
376
377 snprintf(full_path, sizeof(full_path), "%s\\%s",
378 dir, FindFileData.cFileName);
379
380 if (!(FindFileData.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY))
381 {
382 file_queue_put(full_path);
383 }
384 else if (recursive &&
385 strcmp(FindFileData.cFileName, ".") != 0 &&
386 strcmp(FindFileData.cFileName, "..") != 0)
387 {
388 scan_dir(full_path, recursive, start_time);
389 }
390
391 } while (FindNextFile(hFind, &FindFileData));
392
393 FindClose(hFind);
394 }
395}
